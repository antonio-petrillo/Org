#+title: Binary Search Tree
#+filetags: :programming:datastructure:

* Definition
Is a [[id:521327ac-f3c5-42a9-a004-6edfccf6b29a][Tree]] with at most $2$ child per node.
* Formal Definition
$T \in BST ~ class$ if and only if:
$$T.root \gt T.left.root$$
$$T.root \lt T.right.root$$
$$T.left \in BST$$
$$T.right \in BST$$
$$\emptyset \in BST$$
$${T.root} \cap T.left \cap T.right = \emptyset$$
$${T.root} \cup T.left \cup T.right = T$$

In the insertion I will not consider the ~type value~.

           node->left = remove_helper(node->left, key);
       } else if(root->value < key){
           node->right = remove_helper(node->right, key);

       }

node delete_root(node root){
   if (root->left == NULL || root->right == NULL){
       node tmp = node;
       if(root->left == NULL){
           root = root->right;
       } else {
           root = root->left;
       }
       free(tmp);
   }else {
       node min_on_right = minimum_deconnection(root, root->right);
       root->key = min_on_right->key;
       root->value = min_on_right->value;
       free(min_on_right);
   }
   return node;
}

node minimum_deconnection(node parent, node child){
    if(child->left == NULL){
        if(parent->left == child){
            parent->left = child->right;
        }else{
            parent->right = child->right;
        }
        return child;
    } else {
        node tmp = minimum_deconnection(child, child->left);
        return tmp;
    }
}

type insert(bst t, int key){
    node iter = t->root;
    type value = NULL;
    while(iter != NULL){
        if (iter->value > key){
            iter = iter->left;
        }else if (iter->value < key){
            iter = iter->right;
        }else{
            value = iter->value;
            break;
        }

int successor_helper(node root, int key){
    node candidate = NULL;
    node iter = root;
    while (iter != NULL && iter->key != key){
        if (iter->key < key){
            iter = iter->right;
        }else{
            candidate = iter;
            iter = iter->left;
        }
    }
    return candidate;
}

int successor(bst t, int key){
    return successor_helper(t->root, key);

static bool is_bst_helper(node root, int lower_bound, int upper_bound){
    bool result = true;
    if(root != NULL){
        if (root->value > lower_bound && root->value < upper_bound){
            result = is_bst_helper(root->left, lower_bound, root->value);
            if (result){
                result = is_bst_helper(root->right, root->value, upper_bound);
            }
        } else {
            result = false;
        }
    }
    return result;
}

void iterative_in_order_visit(bst t){
    node current = t->root;
    node previous = NULL;
    stack stack_nodes = new_stack();
    while (current != NULL || !is_empty(stack_nodes)){
        if (current != NULL){
           push(&stack_nodes, current);
           current = current->left;
        } else {
           current = top(stack_nodes);
           if (current->right != NULL && previous != current->right){
               visit_operation(current);
               current = current->right;
           } else {
               if (current->right == NULL){
                   visit_operation(current);
               }
               pop(&stack_nodes);
               previous = current;
               current = NULL;
           }
        }
    }
}

void breadth_visit(bst t){
    queue q = new_queue();
    enqueue(&q, t->root);
    while(!is_empty(q)){
       node root = dequeue(&q);
       visit_operation(root);
       if(root->left != NULL){
          enqueue(&q, root->left);
       }
       if(root->right != NULL){
          enqueue(&q, root->right);
       }
    }
}
